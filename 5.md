# Ensure the script runs continuously until manually stopped
Write-Host "Starting Chrome Tab Switcher (2-minute intervals)... Press 'Ctrl + C' in this window to stop." -ForegroundColor Cyan

# Load the required Windows Forms assembly to simulate keystrokes
Add-Type -AssemblyName System.Windows.Forms

# Infinite loop to keep the switcher running
while ($true) {
    # 1. Find the Chrome process
    $chrome = Get-Process -Name "chrome" -ErrorAction SilentlyContinue | Where-Object { $_.MainWindowTitle } | Select-Object -First 1

    if ($chrome) {
        # 2. Bring Chrome to the foreground using Windows API
        $sig = '[DllImport("user32.dll")] public static extern bool SetForegroundWindow(IntPtr hWnd);'
        $type = Add-Type -MemberDefinition $sig -Name "Win32SetForegroundWindow" -Namespace "Win32" -PassThru
        [void]$type::SetForegroundWindow($chrome.MainWindowHandle)

        # Give Windows a brief moment (100ms) to focus the window before sending keys
        Start-Sleep -Milliseconds 100

        # 3. Send "Ctrl + Tab" to switch to the next tab
        # "^" represents the Ctrl key, and "{TAB}" represents the Tab key
        [System.Windows.Forms.SendKeys]::SendWait("^{TAB}")
        
        Write-Host "Switched tab at: $(Get-Date -Format 'HH:mm:ss')" -ForegroundColor Green
    } else {
        Write-Host "Chrome is not currently running or no active window was found." -ForegroundColor Yellow
    }

    # 4. Wait for 2 minutes (120 seconds) before switching again
    Start-Sleep -Seconds 120
}
